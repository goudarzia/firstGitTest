echo "# firstGitTest" >> README.md
git init firstGitTest
git add README.md
git commit -m "first commit"
git remote add origin https://github.com/goudarzia/firstGitTest.git
git push -u origin master

git remote add origin https://github.com/goudarzia/firstGitTest.git
